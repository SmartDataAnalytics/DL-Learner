(define-modifier very linear-modifier(0.8))

# crisp values
(define-fuzzy-concept item1_price crisp(0,4,1.5,1.5))

# fuzzy datatype
(define-fuzzy-concept CheapPrice right-shoulder(0,4,1,2))

# (fuzzy) concepts definition
(define-concept CheapItem (and Item (some hasPrice (very CheapPrice))))

(instance item1 (and Item (some hasPrice item1_price)) 1)

# queries
